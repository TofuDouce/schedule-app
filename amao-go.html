/** * 【阿毛之塔 2.0：核心數據引擎】 * 負責處理全域狀態、存檔讀取及屬性連動。 */ // 1. 全域初始狀態 (基礎值) let state = { lvl: 1, // 當前塔層 nrg: 60, // 能量 (上限 60) gold: 100, // 金幣 mood: 100, // 心情 (影響掉寶與事件觸發) // 戰鬥屬性 stats: { atk: 10, // 攻擊 def: 10, // 防禦 spd: 10, // 速度 luk: 10, // 幸運 (影響 1000+ 道具掉率) int: 10 // 智力 (影響技能威力與金幣倍率) }, // 稱號與成就 title: "初出的冒險犬", skills: [], // 已習得技能 inv: [], // 背包數據 (將承載 1000+ 道具) equips: { head: null, body: null, weapon: null, feet: null }, // 系統狀態 isBattle: false, isProcessing: false, lastSave: Date.now() }; const AmaoApp = { // 2. 初始化：從瀏覽器讀取舊進度 init: function() { const saved = localStorage.getItem('amao_tower_v3_master'); if (saved) { try { const parsed = JSON.parse(saved); state = { ...state, ...parsed }; // 合併存檔，保留新版本欄位 this.addLog("✨ 歡迎回來！阿毛已從上次的時空裂縫中歸隊。"); } catch(e) { console.error("存檔損壞，已重置。"); } } this.updateUI(); this.startAutoHeal(); // 啟動自動恢復機制 }, // 3. UI 刷新引擎 (確保一目瞭然) updateUI: function() { document.getElementById('ui-lvl').innerText = state.lvl; document.getElementById('ui-nrg').innerText = state.nrg; document.getElementById('ui-gold').innerText = state.gold; // 能量顏色動態警告 const nrgEl = document.getElementById('ui-nrg'); if (state.nrg <= 10) nrgEl.style.color = "var(--danger)"; else if (state.nrg <= 30) nrgEl.style.color = "var(--accent)"; else nrgEl.style.color = "#3498db"; // 保存到本地 localStorage.setItem('amao_tower_v3_master', JSON.stringify(state)); }, // 4. 極致特大日誌輸出 (防文字太小問題) addLog: function(msg, color = "var(--text-main)") { const logBox = document.getElementById('msg-log'); const entry = document.createElement('div'); entry.className = 'log-entry'; entry.style.color = color; // 自動加上時間戳記 const now = new Date(); const timeStr = `${now.getHours()}:${now.getMinutes()}:${now.getSeconds()}`; entry.innerHTML = `${timeStr}${msg}`; logBox.appendChild(entry); // 限制日誌數量防止卡頓，保留 50 條 if (logBox.childNodes.length > 50) { logBox.removeChild(logBox.firstChild); } }, // 5. 每 3 分鐘自動恢復 1 點能量 (生活化設計) startAutoHeal: function() { setInterval(() => { if (state.nrg < 60) { state.nrg++; this.updateUI(); } if (state.mood < 100) { state.mood = Math.min(100, state.mood + 0.5); } }, 180000); }, // 6. 重啟冒險 (創世重置) reset: function() { if (confirm("🚨 警告：這將抹除所有層數、道具與 5000 種美食紀錄！確定重啟嗎？")) { localStorage.removeItem('amao_tower_v3_master'); location.reload(); } } }; // 立即啟動 window.onload = () => AmaoApp.init();
